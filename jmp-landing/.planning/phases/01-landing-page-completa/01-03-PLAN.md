    ```tsx
    import { useState } from 'react'
    import { Loader2 } from 'lucide-react'
    import { OBRIGADO_PAGE_URL } from '../constants'

    const UF_OPTIONS = [
      'AC','AL','AP','AM','BA','CE','DF','ES','GO',
      'MA','MT','MS','MG','PA','PB','PR','PE','PI',
      'RJ','RN','RS','RO','RR','SC','SP','SE','TO',
    ]

    interface FormData {
      nome: string
      celular: string
      uf: string
      cidade: string
      interesse: string
      rebanho: string
    }

    interface FormErrors {
      nome?: string
      celular?: string
      uf?: string
      cidade?: string
      interesse?: string
      rebanho?: string
    }

    function applyPhoneMask(value: string): string {
      const digits = value.replace(/\D/g, '').slice(0, 11)
      if (digits.length <= 2) return digits.length ? `(${digits}` : ''
      if (digits.length <= 7) return `(${digits.slice(0,2)}) ${digits.slice(2)}`
      return `(${digits.slice(0,2)}) ${digits.slice(2,7)}-${digits.slice(7)}`
    }

    function validate(data: FormData): FormErrors {
      const errors: FormErrors = {}
      if (!data.nome.trim()) errors.nome = 'Nome é obrigatório'
      const digits = data.celular.replace(/\D/g, '')
      if (digits.length < 10) errors.celular = 'Celular inválido — informe DDD + número'
      if (!data.uf) errors.uf = 'UF é obrigatória'
      if (!data.cidade.trim()) errors.cidade = 'Cidade é obrigatória'
      if (!data.interesse) errors.interesse = 'Selecione seu interesse'
      if (!data.rebanho) errors.rebanho = 'Selecione o tamanho do rebanho'
      return errors
    }

    // handleSubmit — isolated integration point for future Google Sheets / CRM
    // Replace the body of this function to integrate with external services
    async function handleSubmit(data: FormData): Promise<void> {
      // TODO: POST to Google Sheets API or Supabase here
      console.log('Lead captured:', data)
      await new Promise(resolve => setTimeout(resolve, 800))
    }

    export function Form() {
      const [formData, setFormData] = useState<FormData>({
        nome: '', celular: '', uf: '', cidade: '', interesse: '', rebanho: '',
      })
      const [errors, setErrors] = useState<FormErrors>({})
      const [loading, setLoading] = useState(false)

      function handleChange(field: keyof FormData, value: string) {
        setFormData(prev => ({ ...prev, [field]: value }))
        if (errors[field]) setErrors(prev => ({ ...prev, [field]: undefined }))
      }

      async function onSubmit(e: React.FormEvent) {
        e.preventDefault()
        const validationErrors = validate(formData)
        if (Object.keys(validationErrors).length > 0) {
          setErrors(validationErrors)
          return
        }
        setLoading(true)
        try {
          await handleSubmit(formData)
          // Redirect to thank-you page — /obrigado-jmp.html handles WhatsApp redirect with 5s countdown
          window.location.href = OBRIGADO_PAGE_URL
        } finally {
          setLoading(false)
        }
      }

      const inputClass = (field: keyof FormErrors) =>
        `w-full px-3 py-2.5 rounded-md border text-sm transition-colors focus:outline-none focus:ring-2 focus:ring-gold/50 ${
          errors[field]
            ? 'border-red-500 bg-red-50'
            : 'border-black/15 bg-white hover:border-black/30 focus:border-gold'
        }`

      return (
        <section id="inscricao" className="py-16 px-4 bg-white">
          <div className="max-w-lg mx-auto">
            <div className="text-center mb-8">
              <h2 className="text-3xl font-black mb-2">Inscreva-se no Leilão</h2>
              <p className="text-black/60">Preencha o formulário e acesse o grupo exclusivo no WhatsApp</p>
            </div>

            <form onSubmit={onSubmit} noValidate className="space-y-4">
              {/* Nome Completo */}
              <div>
                <label className="block text-sm font-semibold mb-1">Nome Completo *</label>
                <input
                  type="text"
                  value={formData.nome}
                  onChange={e => handleChange('nome', e.target.value)}
                  placeholder="Seu nome completo"
                  className={inputClass('nome')}
                />
                {errors.nome && <p className="text-red-500 text-xs mt-1">{errors.nome}</p>}
              </div>

              {/* Celular */}
              <div>
                <label className="block text-sm font-semibold mb-1">Celular *</label>
                <input
                  type="tel"
                  value={formData.celular}
                  onChange={e => handleChange('celular', applyPhoneMask(e.target.value))}
                  placeholder="(00) 00000-0000"
                  className={inputClass('celular')}
                />
                {errors.celular && <p className="text-red-500 text-xs mt-1">{errors.celular}</p>}
              </div>

              {/* UF + Cidade (row) */}
              <div className="grid grid-cols-2 gap-3">
                <div>
                  <label className="block text-sm font-semibold mb-1">UF *</label>
                  <select
                    value={formData.uf}
                    onChange={e => handleChange('uf', e.target.value)}
                    className={inputClass('uf')}
                  >
                    <option value="">Selecione</option>
                    {UF_OPTIONS.map(uf => (
                      <option key={uf} value={uf}>{uf}</option>
                    ))}
                  </select>
                  {errors.uf && <p className="text-red-500 text-xs mt-1">{errors.uf}</p>}
                </div>
                <div>
                  <label className="block text-sm font-semibold mb-1">Cidade *</label>
                  <input
                    type="text"
                    value={formData.cidade}
                    onChange={e => handleChange('cidade', e.target.value)}
                    placeholder="Sua cidade"
                    className={inputClass('cidade')}
                  />
                  {errors.cidade && <p className="text-red-500 text-xs mt-1">{errors.cidade}</p>}
                </div>
              </div>

              {/* Interesse */}
              <div>
                <label className="block text-sm font-semibold mb-1">Interesse *</label>
                <select
                  value={formData.interesse}
                  onChange={e => handleChange('interesse', e.target.value)}
                  className={inputClass('interesse')}
                >
                  <option value="">Selecione seu interesse</option>
                  <option value="Touros">Touros</option>
                  <option value="Matrizes">Matrizes</option>
                  <option value="Embrião">Embrião</option>
                  <option value="Sêmen">Sêmen</option>
                </select>
                {errors.interesse && <p className="text-red-500 text-xs mt-1">{errors.interesse}</p>}
              </div>

              {/* Tamanho do Rebanho */}
              <div>
                <label className="block text-sm font-semibold mb-1">Tamanho do Rebanho *</label>
                <select
                  value={formData.rebanho}
                  onChange={e => handleChange('rebanho', e.target.value)}
                  className={inputClass('rebanho')}
                >
                  <option value="">Selecione</option>
                  <option value="Até 50 cabeças">Até 50 cabeças</option>
                  <option value="51–200">51–200 cabeças</option>
                  <option value="201–500">201–500 cabeças</option>
                  <option value="501–1.000">501–1.000 cabeças</option>
                  <option value="Acima de 1.000">Acima de 1.000 cabeças</option>
                </select>
                {errors.rebanho && <p className="text-red-500 text-xs mt-1">{errors.rebanho}</p>}
              </div>

              {/* Submit */}
              <button
                type="submit"
                disabled={loading}
                className="w-full bg-gold hover:bg-gold-dark disabled:opacity-60 text-white font-bold py-3 px-6 rounded-md transition-all duration-200 hover:-translate-y-0.5 hover:shadow-lg flex items-center justify-center gap-2 mt-2"
              >
                {loading ? (
                  <>
                    <Loader2 className="w-5 h-5 animate-spin" />
                    Enviando…
                  </>
                ) : (
                  'Quero me inscrever'
                )}
              </button>
            </form>
          </div>
        </section>
      )
    }
    ```
